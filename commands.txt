>git --version [command for checking git version]
>git config --list [git configuration list]
>git config --global user.name "username"[command for adding username]
>git config --global user.email "useremail@gmail.com"[command for adding email]
>git init [initializing git in present working directory]
>git status [list all untracked files]
>git add -A [command for adding all untracked files to staging area]
>git commit -m "commit message" [adding staged files to local repository]
>git log [command for listinh commit history]
>git remote add name remote-path
>git push remote-name master
>git branch
